<script>
let password = "25012006";

function checkPass() {
  let val = document.getElementById("pass").value;
  if (val === password) {
    next(1);
  } else {
    alert("Wrong password 😤");
  }
}

function next(level) {
  document.querySelectorAll(".screen").forEach(s => s.classList.remove("active"));
  document.getElementById("l" + level).classList.add("active");
  document.getElementById("level").innerText = "Level " + level;

  if (level === 3) startStory();
}

// STORY TEXT
function startStory() {
  let text = "From Mysha... to Adhara... to Dhanvantari ji... to Rasmalai Sherni 🐯... 2 years and still not bored of me 😭";
  let i = 0;
  let el = document.getElementById("story");
  el.innerHTML = "";

  function type() {
    if (i < text.length) {
      el.innerHTML += text.charAt(i);
      i++;
      setTimeout(type, 40);
    }
  }
  type();
}

// YES BUTTON ESCAPE
let yes = document.getElementById("yes");
let tries = 0;

yes.addEventListener("mouseover", () => {
  if (tries < 6) {
    yes.style.top = Math.random() * window.innerHeight + "px";
    yes.style.left = Math.random() * window.innerWidth + "px";
    tries++;
  } else {
    yes.disabled = true;
    yes.innerText = "Not Possible 😈";
  }
});

yes.addEventListener("click", () => {
  alert("Galat choice 😡");
});
